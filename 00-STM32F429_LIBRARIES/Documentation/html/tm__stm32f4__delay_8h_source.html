173 static __INLINE void Delay(uint32_t micros) {
174 #if defined(TM_DELAY_TIM)
175  volatile uint32_t timer = TM_DELAY_TIM->CNT;
176 
177  do {
178  /* Count timer ticks */
179  while ((TM_DELAY_TIM->CNT - timer) == 0);
180 
181  /* Increase timer */
182  timer = TM_DELAY_TIM->CNT;
183 
184  /* Decrease microseconds */
185  } while (--micros);
186 #else
187  uint32_t amicros;
188 
189  /* Multiply micro seconds */
190  amicros = (micros) * (mult);
191 
192  #ifdef __GNUC__
193  if (SystemCoreClock == 180000000 || SystemCoreClock == 100000000) {
194  amicros -= mult;
195  }
196  #endif
197 
198  /* If clock is 100MHz, then add additional multiplier */
199  /* 100/3 = 33.3 = 33 and delay wouldn't be so accurate */
200  #if defined(STM32F411xE)
201  amicros += mult;
202  #endif
203 
204  /* While loop */
205  while (amicros--);
206 #endif /* TM_DELAY_TIM */
207 }
